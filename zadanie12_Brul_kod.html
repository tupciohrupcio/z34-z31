<body>
    <h2>Bruliński Olek Zadanie 12</h2>

    <label>Dzień 1 (D1_Brul): <input type="number" id="D1_Brul"></label><br>
    <label>Godzina 1 (G1_Brul): <input type="number" id="G1_Brul"></label><br>
    <label>Dzień 2 (D2_Brul): <input type="number" id="D2_Brul"></label><br>
    <label>Godzina 2 (G2_Brul): <input type="number" id="G2_Brul"></label><br><br>

    <button onclick="obliczSekundy()">Oblicz ile sekund</button>

    <p id="wynik"></p>

    <script>
        function obliczSekundy() {
            let D1 = parseInt(document.getElementById('D1_Brul').value);
            let G1 = parseInt(document.getElementById('G1_Brul').value);
            let D2 = parseInt(document.getElementById('D2_Brul').value);
            let G2 = parseInt(document.getElementById('G2_Brul').value);

            let godziny1 = D1 * 24 + G1;
            let godziny2 = D2 * 24 + G2;
            let roznicaGodzin = godziny2 - godziny1;

            let sekundy = roznicaGodzin * 3600;

            document.getElementById('wynik').innerText = `Ilość sekund: ${sekundy} sekund`;
        }
    </script>
