Zadanie – Programowanie obiektowe

Tutaj pojawi się wynik działania metody obiektu.

Kod źródłowy

var Samochod = {
    kolor: "czerwony",
    marka: "BMW",
    rok: 2020,

    samochod_wyswietl_brulinski: function () {
        document.getElementById("wynik").innerHTML =
            "Nazwa obiektu: Samochod" +
            "Pola: kolor, marka, rok" +
            "Metoda: samochod_wyswietl_brulinski";
    }
};
