        function rozkladEgipski() {
            let m = 4;
            let wynik = "Rozkład ułamka 1/" + m + " na trzy ułamki egipskie:<br>";

            let a = 6;
            let b = 30;
            let c = 20;

            let suma = 1/a + 1/b + 1/c;
            let oryginal = 1/m;

            wynik += `1/${m} = 1/${a} + 1/${b} + 1/${c}<br>`;
            wynik += `Suma po prawej stronie = ${suma.toFixed(5)}<br>`;
            wynik += `Oryginalny ułamek = ${oryginal.toFixed(5)}<br>`;

            if (Math.abs(suma - oryginal) < 0.00001) {
                wynik += `<b style="color:green;">✅ Rozkład poprawny!</b>`;
            } else {
                wynik += `<b style="color:red;">❌ Rozkład błędny!</b>`;
            }

            document.getElementById("wynik").innerHTML = wynik;
        }
    </script>
